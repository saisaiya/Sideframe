direction属性值为right,此时为右侧边菜单。toggle为true，单点菜单可以自由操控，有下拉和上拉两个动作。doubleTapToGo为true,菜单之间不发生互斥，可以同时打开菜单。

CODE
$(function(){
    $(".slideGroup slideGroup1").sideframe({
    direction:"right",
    toggle:true,
    doubleTapToGo:true,
  });
 });
